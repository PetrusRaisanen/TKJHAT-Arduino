36void Display::hspan(int16_t x1, int16_t x2, int16_t y) {
37 if (y < 0 || y >= (int16_t)_disp.height) return;
38 if (x1 > x2) { int16_t t = x1; x1 = x2; x2 = t; }
39 if (x2 < 0 || x1 >= (int16_t)_disp.width) return;
40 if (x1 < 0) x1 = 0;
41 if (x2 >= (int16_t)_disp.width) x2 = (int16_t)_disp.width - 1;
42
43 for (int16_t x = x1; x <= x2; ++x) {
44 ssd1306_draw_pixel(&_disp, (uint32_t)x, (uint32_t)y);
45 }

49void Display::drawCircle(int16_t x0, int16_t y0, int16_t r, bool fill) {
50 // Draw a circle using the Bresenham algorithm
51 if (r < 0)
52 return;
53 if (r == 0) {
54 putp(x0, y0);
55 ssd1306_show(&_disp);
56 return;
57 }
58
59 // Midpoint circle algorithm
60 int16_t f = 1 - r;
61 int16_t ddF_x = 1;
62 int16_t ddF_y = -2 * r;
63 int16_t x = 0;
64 int16_t y = r;
65
66 if (fill) {
67 hspan((int16_t)(x0 - r), (int16_t)(x0 + r), y0); // center row
68 } else {
69 putp(x0, (int16_t)(y0 + r));
70 putp(x0, (int16_t)(y0 - r));
71 putp((int16_t)(x0 + r), y0);
72 putp((int16_t)(x0 - r), y0);
73 }
74
75 while (x < y) {
76 if (f >= 0) { y--; ddF_y += 2; f += ddF_y; }
77 x++; ddF_x += 2; f += ddF_x;
78
79 if (fill) {
80 hspan((int16_t)(x0 - x), (int16_t)(x0 + x), (int16_t)(y0 + y));
81 hspan((int16_t)(x0 - x), (int16_t)(x0 + x), (int16_t)(y0 - y));
82 hspan((int16_t)(x0 - y), (int16_t)(x0 + y), (int16_t)(y0 + x));
83 hspan((int16_t)(x0 - y), (int16_t)(x0 + y), (int16_t)(y0 - x));
84 } else {
85 putp((int16_t)(x0 + x), (int16_t)(y0 + y));
86 putp((int16_t)(x0 - x), (int16_t)(y0 + y));
87 putp((int16_t)(x0 + x), (int16_t)(y0 - y));
88 putp((int16_t)(x0 - x), (int16_t)(y0 - y));
89 putp((int16_t)(x0 + y), (int16_t)(y0 + x));
90 putp((int16_t)(x0 - y), (int16_t)(y0 + x));
91 putp((int16_t)(x0 + y), (int16_t)(y0 - x));
92 putp((int16_t)(x0 - y), (int16_t)(y0 - x));
93 }
94 }
95 ssd1306_show(&_disp); // remove if batching multiple draws
96}
